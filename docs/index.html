async function loadSmartLists() {
  let smartLists = JSON.parse(
    localStorage.getItem(SMART_LISTS_KEY) || "null"
  );

  if (!smartLists) {
    const res = await fetch("./Smart Lists.json", {
      cache: "no-store"
    });
    smartLists = await res.json();

    localStorage.setItem(
      SMART_LISTS_KEY,
      JSON.stringify(smartLists)
    );

    console.log("Smart Lists loaded from GitHub");
  } else {
    console.log("Smart Lists loaded from localStorage");
  }

  return smartLists;
}

function validateSmartListsMembers(smartLists, items) {
  const validUUIDs = new Set(items.map(item => item.UUID));
  let changed = false;

  smartLists.forEach(listObj => {
    if (!Array.isArray(listObj.members)) return;

    listObj.members = listObj.members.filter(uuid => {
      const exists = validUUIDs.has(uuid);
      if (!exists) {
        console.log(
          "Deleted Item from",
          listObj.list
        );
        changed = true;
      }
      return exists;
    });
  });

  return changed;
}

function saveSmartLists(smartLists) {
  localStorage.setItem(
    SMART_LISTS_KEY,
    JSON.stringify(smartLists)
  );

  console.log("Updated Smart Lists:");
  console.log(JSON.stringify(smartLists, null, 2));
}

async function prepareSmartLists(items) {
  const smartLists = await loadSmartLists();

  const changed = validateSmartListsMembers(
    smartLists,
    items
  );

  if (changed) {
    saveSmartLists(smartLists);
  } else {
    console.log("Smart Lists unchanged");
    console.log(JSON.stringify(smartLists, null, 2));
  }

  return smartLists;
}

}

function norm(v) {
  return (v ?? "").toString().trim();
}

function getItemDomain(i) {
  return norm(i.domain || i.Domain);
}

function getItemRegion(i) {
  return norm(i.region || i.Region);
}

function uniqueSorted(arr) {
  return [...new Set(arr.filter(v => v && v !== "N/A"))].sort((a,b) =>
    a.localeCompare(b, undefined, { sensitivity:"base" })
  );
}

function buildSuggestionList(container, values, onPick) {
  container.innerHTML = "";
  values.forEach(val => {
    const div = document.createElement("div");
    div.textContent = val;
    div.style.padding = "12px 16px";
    div.style.cursor = "pointer";
    div.onclick = () => onPick(val);
    container.appendChild(div);
  });
  container.style.display = values.length ? "flex" : "none";
}

function domainsFromItems() {
  return uniqueSorted(items.map(getItemDomain));
}

function regionsForDomain(domain) {
  const d = norm(domain).toLowerCase();
  if (!d) return [];
  return uniqueSorted(
    items
      .filter(i => getItemDomain(i).toLowerCase() === d)
      .map(getItemRegion)
  );
}

// DOMAIN input autocomplete
domainInput.oninput = () => {
  const typed = norm(domainInput.value);
  newBottleDomain = typed;

  // If domain changes, clear region (because it is dependent)
  newBottleRegion = "";
  regionInput.value = "";
  regionSuggestions.style.display = "none";
  regionInput.placeholder = typed ? "Enter region" : "Select a domain first";

  const q = typed.toLowerCase();
  if (!q) {
    domainSuggestions.style.display = "none";
    return;
  }

  const matches = domainsFromItems().filter(d =>
    d.toLowerCase().includes(q)
  );

  buildSuggestionList(domainSuggestions, matches, picked => {
    domainInput.value = picked;
    newBottleDomain = picked;

    // Clear region on pick too, then unlock region placeholder
    newBottleRegion = "";
    regionInput.value = "";
    regionInput.placeholder = "Enter region";

    domainSuggestions.style.display = "none";
  });
};

// REGION input autocomplete (depends on domain)
regionInput.oninput = () => {
  const domain = norm(domainInput.value);
  newBottleDomain = domain;

  // must have domain to suggest regions
  if (!domain) {
    regionSuggestions.style.display = "none";
    regionInput.placeholder = "Select a domain first";
    newBottleRegion = norm(regionInput.value);
    return;
  }

  const typed = norm(regionInput.value);
  newBottleRegion = typed;

  const q = typed.toLowerCase();
  const pool = regionsForDomain(domain);

  if (!q) {
    // If they haven‚Äôt typed yet, show the full list for that domain
    buildSuggestionList(regionSuggestions, pool, picked => {
      regionInput.value = picked;
      newBottleRegion = picked;
      regionSuggestions.style.display = "none";
    });
    return;
  }

  const matches = pool.filter(r => r.toLowerCase().includes(q));
  buildSuggestionList(regionSuggestions, matches, picked => {
    regionInput.value = picked;
    newBottleRegion = picked;
    regionSuggestions.style.display = "none";
  });
};

  const sendData = () => {
    printWindow.printItems = payload;
    printWindow.sourcePageTitle = document.title;
  };

  if (printWindow.document.readyState === "complete") {
    sendData();
  } else {
    printWindow.onload = sendData;
  }
}

function buildListEditorData(targetList, items) {
  const smartLists = JSON.parse(
    localStorage.getItem("Smart Lists") || "[]"
  );

  const match = smartLists.find(
    l => l.list === targetList
  );

  let currentList = [];
  const validUUIDs = new Set(items.map(i => i.UUID));

  if (match && Array.isArray(match.members)) {
    match.members.forEach(uuid => {
      if (validUUIDs.has(uuid)) {
        currentList.push(uuid);
      }
    });
  }

  const currentSet = new Set(currentList);

  const suggestedList = items
    .map(i => i.UUID)
    .filter(uuid => !currentSet.has(uuid));

  return { currentList, suggestedList, smartLists };
}

function openListEditorPopup() {
  const {
    currentList,
    suggestedList,
    smartLists
  } = buildListEditorData(targetList, items);

  let tempCurrent = [...currentList];
  let tempSuggested = [...suggestedList];

  /* ---------- OVERLAY ---------- */
  const overlay = document.createElement("div");
  overlay.className = "overlay show";

  /* ---------- POPUP ---------- */
  const popup = document.createElement("div");
  popup.style.cssText = `
    position:fixed;
    inset:20px;
    background:white;
    border-radius:16px;
    padding:16px;
    overflow:auto;
    z-index:400;
  `;

  popup.innerHTML = `
    

      ${targetList}
    

  `;

  /* ---------- CURRENT LIST ---------- */
  const currentWrap = document.createElement("div");
  popup.appendChild(currentWrap);

  function renderCurrentList() {
    currentWrap.innerHTML = "";

    tempCurrent.forEach((uuid, index) => {
      const item = items.find(i => i.UUID === uuid);
      if (!item) return;

      const row = document.createElement("div");
      row.style.cssText = `
        display:flex;
        align-items:center;
        gap:8px;
        padding:8px;
        margin-bottom:6px;
        background:#f2f2f2;
        border-radius:8px;
        cursor:grab;
      `;

      const title = document.createElement("div");
      title.textContent = smartItemTitle(item);
      title.style.flex = "1";

      const del = document.createElement("div");
      del.textContent = "‚úï";
      del.style.cursor = "pointer";

      del.onclick = () => {
        tempCurrent.splice(index, 1);
        tempSuggested.push(uuid);
        renderCurrentList();
        renderSuggestions();
      };

      row.appendChild(title);
      row.appendChild(del);
      currentWrap.appendChild(row);

      /* üîπ Reorder (long press move up/down) */
      let hold;
      row.onpointerdown = () => {
        hold = setTimeout(() => {
          if (index > 0) {
            [tempCurrent[index - 1], tempCurrent[index]] =
              [tempCurrent[index], tempCurrent[index - 1]];
            renderCurrentList();
          }
        }, 600);
      };
      row.onpointerup = () => clearTimeout(hold);
      row.onpointerleave = () => clearTimeout(hold);
    });
  }

  /* ---------- SUGGEST INPUT ---------- */
  const input = document.createElement("input");
  input.placeholder = "Add item‚Ä¶";
  input.style.cssText = `
    width:100%;
    padding:10px;
    margin:10px 0;
  `;

  const datalist = document.createElement("datalist");
  datalist.id = "suggestions";

  function renderSuggestions() {
    datalist.innerHTML = "";
    tempSuggested.forEach(uuid => {
      const item = items.find(i => i.UUID === uuid);
      if (!item) return;
      const opt = document.createElement("option");
      opt.value = smartItemTitle(item);
      datalist.appendChild(opt);
    });
  }

  input.setAttribute("list", "suggestions");

  input.onchange = () => {
    const match = tempSuggested.find(uuid => {
      const item = items.find(i => i.UUID === uuid);
      return item && smartItemTitle(item) === input.value;
    });

    if (match) {
      tempCurrent.push(match);
      tempSuggested = tempSuggested.filter(u => u !== match);
      input.value = "";
      renderCurrentList();
      renderSuggestions();
    }
  };

  popup.appendChild(input);
  popup.appendChild(datalist);

  /* ---------- DELETE LIST ---------- */
  const deleteBtn = document.createElement("button");
  deleteBtn.textContent = "Delete List";
  deleteBtn.style.cssText = `
    margin-top:10px;
    color:#7b1e1e;
  `;

 deleteBtn.onclick = async () => {
  if (!confirm("Delete this list?")) return;

  const updated = smartLists.filter(l => l.list !== targetList);

  localStorage.setItem("Smart Lists", JSON.stringify(updated));

  console.log("Updated Smart Lists:");
  console.log(updated);

  close();

  // ‚úÖ re-render after deletion
  updateGlobalFilters();
  await renderItems();
};


  popup.appendChild(deleteBtn);

  /* ---------- CONFIRM ---------- */
  const confirmBtn = document.createElement("button");
  confirmBtn.textContent = "Confirm";
  confirmBtn.style.cssText = `
    margin-top:12px;
    background:#1f5e3b;
    color:white;
    padding:10px;
    border-radius:10px;
    border:none;
    width:100%;
  `;

  confirmBtn.onclick = async () => {
  const filtered = smartLists.filter(l => l.list !== targetList);

  filtered.push({
    list: targetList,
    members: tempCurrent
  });

  localStorage.setItem("Smart Lists", JSON.stringify(filtered));

  console.log("Updated Smart Lists:");
  console.log(filtered);

  close();

  // ‚úÖ re-render with latest smart list changes
  updateGlobalFilters();
  await renderItems();
};

  popup.appendChild(confirmBtn);

  document.body.appendChild(overlay);
  document.body.appendChild(popup);

  renderCurrentList();
  renderSuggestions();

  function close() {
    overlay.remove();
    popup.remove();
  }
}

function updateFloatingSourceVisibility() {
  if (!floatingSourceEl || !tabsEl) return;

  const tabsRect = tabsEl.getBoundingClientRect();

  // Show once the tabs have scrolled past the top edge
  if (tabsRect.bottom <= 10) {
    floatingSourceEl.classList.add("show");
  } else {
    floatingSourceEl.classList.remove("show");
  }
}
